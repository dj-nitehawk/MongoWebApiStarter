UBUNTU SERVER SETUP [tested on ubuntu server 20.10]
===================

UPDATE PKGS [login as root]
-----------
apt-get update
apt-get upgrade

SETUP ACCOUNT FOR APPLICATION
-----------------------------
adduser MongoWebApiStarter

INSTALL DOTNET 6 RUNTIME
---------------------------
wget https://packages.microsoft.com/config/ubuntu/20.10/packages-microsoft-prod.deb -O packages-microsoft-prod.deb
dpkg -i packages-microsoft-prod.deb
rm packages-microsoft-prod.deb
apt-get update
apt-get install -y apt-transport-https aspnetcore-runtime-6.0
dotnet --info

INSTALL MONGODB SERVER
----------------------
nano /etc/sysctl.conf
	[add following line:]
	vm.swappiness = 1
sysctl -p
wget -qO - https://www.mongodb.org/static/pgp/server-5.0.asc | sudo apt-key add -
echo "deb [ arch=amd64,arm64 ] https://repo.mongodb.org/apt/ubuntu focal/mongodb-org/5.0 multiverse" | sudo tee /etc/apt/sources.list.d/mongodb-org-5.0.list
apt-get update
apt-get install -y mongodb-org
mkdir /mongodb
mkdir /mongodb/log
chown -R mongodb:mongodb /mongodb
echo "" > /etc/mongod.conf
nano /etc/mongod.conf
	[copy+paste from "mongod.conf"]
systemctl enable mongod.service
service mongod start
service mongod status
mongosh --eval "rs.initiate()"
mongosh --eval "rs.status()"

SETUP APP SECRETS
-----------------
nano /etc/environment
	[copy+paste from "env-variables.txt"]
	[set the correct secrets]
shutdown -r 0

DEPLOY .NET CORE APP [login with MongoWebApiStarter account]
--------------------
mkdir /home/MongoWebApiStarter/app
	[copy the files over via sFTP]
cd /home/MongoWebApiStarter/app
sudo chmod +x MongoWebApiStarter.dll
dotnet ./MongoWebApiStarter.dll
exit

INSTALL NGINX WITH LETSENCRYPT SSL [login as root]
----------------------------------
apt-get install -y software-properties-common certbot python3-certbot-nginx nginx
systemctl start nginx
systemctl status nginx
echo "" > /etc/nginx/sites-available/default
nano /etc/nginx/sites-available/default
	[copy+paste from "nginx.conf"]
certbot certonly -d MongoWebApiStarter.com -i nginx
certbot renew --dry-run
systemctl list-timers
nginx -t
systemctl restart nginx
systemctl status nginx

INSTALL KESTRAL SERVICE
-----------------------
nano /etc/systemd/system/MongoWebApiStarter.service
	[copy+paste from "MongoWebApiStarter.service"]
systemctl enable MongoWebApiStarter.service
systemctl start MongoWebApiStarter.service
systemctl status MongoWebApiStarter.service

CONFIGURE FIREWALL
------------------
apt-get install -y ufw
ufw reset

ufw allow 22
ufw allow 80
ufw allow 443
ufw deny from any

ufw enable

ufw status

RESTART
-------
shutdown -r 0
